# 进制转换
十进制转其他进制 (最多到16进制)
```js
let ten2Another = function (num, base) {
  let arr = []
  let marks = '0123456789ABCDE' // 所以最多到 16进制
  let result = ''
  while (num > 0) {
    arr.push(num % base)
    num = Math.floor(num / base)
  }
  arr.reduceRight((pre, cur) => {
    result += marks.charAt(cur)
  }, '')
  return result
}

console.log(ten2Another(19, 2) )
```
